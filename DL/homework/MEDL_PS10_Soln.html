# Your code here

encoder = tf.keras.models.Sequential([
    tf.keras.layers.Conv2D(48, (3,3), activation = 'relu', padding = 'SAME', input_shape = (64, 64, 3)),
    tf.keras.layers.MaxPool2D((2, 2), padding = 'SAME'),
    
    tf.keras.layers.Conv2D(96, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.MaxPool2D((2,2), padding = 'SAME'),
    
    tf.keras.layers.Conv2D(192, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.MaxPool2D((2,2), padding = 'SAME'),
    
    tf.keras.layers.Conv2D(32, (1,1), activation = 'relu', padding = 'SAME')
])

# Latent space
latentSize = (8,8,32)

decoder = tf.keras.models.Sequential([
    tf.keras.layers.Conv2D(192, (1,1), activation = 'relu', padding = 'SAME', input_shape = (latentSize)),
    tf.keras.layers.UpSampling2D((2,2)),
    
    tf.keras.layers.Conv2D(192, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.UpSampling2D((2,2)),
    
    tf.keras.layers.Conv2D(96, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.UpSampling2D((2,2)),
    
    tf.keras.layers.Conv2D(48, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.Conv2D(3, (3,3), activation = 'relu', padding = 'SAME')
])

autoencoder = tf.keras.models.Sequential([encoder, decoder])

(6) Explain the reason for the difference between the two images in terms of latent space.

Answer)

The first image is simply an average of the pixels of the white dog and the black dog, so the two images overlap.

The second image is the average of the encoded features of a white dog and a black dog, and a dog with new features is generated.
