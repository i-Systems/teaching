TIME_STEPS = 288

def create_sequences(values, time_steps = TIME_STEPS):
    output = []
    for i in range(len(values) - time_steps + 1 ):
        output.append(values[i: (i + time_steps)])
    return np.stack(output)

def autoencoder_model(X):
    
    L1 = tf.keras.layers.LSTM(16, 
                              input_shape = (X.shape[1], X.shape[2]), 
                              activation = 'relu', 
                              return_sequences = True)
    
    L2 = tf.keras.layers.LSTM(4,
                              activation = 'relu',
                              return_sequences = False)
    
    L3 = tf.keras.layers.RepeatVector(X.shape[1])
    
    L4 = tf.keras.layers.LSTM(4,
                              activation = 'relu',
                              return_sequences = True)
    
    L5 = tf.keras.layers.LSTM(16,
                              activation = 'relu',
                              return_sequences = True)
    
    output = tf.keras.layers.TimeDistributed(tf.keras.layers.Dense(X.shape[2]))
    
    model = tf.keras.Sequential([L1, L2, L3, L4, L5, output])
    
    return model
